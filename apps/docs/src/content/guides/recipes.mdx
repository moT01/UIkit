The component playground (`/`) shows each component in isolation.
This guide collects the patterns that compose two or more.

## Form composition

```tsx
import { FormGroup, Input, HelpBlock, Button } from '@freecodecamp/uikit';

export function NewsletterForm() {
  return (
    <form onSubmit={onSubmit}>
      <FormGroup>
        <label htmlFor='email'>Email address</label>
        <Input id='email' type='email' required />
        <HelpBlock>One curriculum update per week.</HelpBlock>
      </FormGroup>
      <Button type='submit' variant='cta'>
        Subscribe
      </Button>
    </form>
  );
}
```

## Toast on action

```tsx
import { useToast, Button } from '@freecodecamp/uikit';

export function SaveButton() {
  const toast = useToast();
  return (
    <Button
      onClick={() => {
        save();
        toast.create({
          title: 'Saved',
          description: 'Your progress is synced.',
          variant: 'success'
        });
      }}
    >
      Save
    </Button>
  );
}
```

## Sidebar layout with navbar

```tsx
import { SidebarLayout, Navbar, Sidebar } from '@freecodecamp/uikit';

export function Shell({ children }) {
  return (
    <SidebarLayout
      header={<Navbar start={<Brand />} end={<UserMenu />} />}
      aside={<Sidebar sections={navSections} />}
    >
      {children}
    </SidebarLayout>
  );
}
```
